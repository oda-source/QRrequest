// フォーム送信処理 document.getElementById('qrRequestForm').addEventListener('submit', function(e) { e.preventDefault(); const submitBtn = document.getElementById('submitBtn'); const gasUrl = localStorage.getItem('gasWebApiUrl'); if (!gasUrl) { alert('【エラー】GASのURLが設定されていません。\n右上の「⚙️ GAS設定」からURLを入力してください。'); return; } // URLパラメータから自動取得したボードURLとボード名を取り出す const urlParams = new URLSearchParams(window.location.search); const boardUrl = urlParams.get('boardUrl') || ''; const boardName = urlParams.get('boardName') || ''; submitBtn.disabled = true; submitBtn.textContent = '送信中...'; const formData = new FormData(this); const data = { email: formData.get('email'), courseNumbers: formData.get('courseNumbers'), boardUrl: boardUrl, boardName: boardName }; fetch(gasUrl, { method: 'POST', mode: 'no-cors', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(data) }) .then(() => { alert('依頼が正常に送信されました！'); document.getElementById('qrRequestForm').reset(); }) .catch(err => { console.error(err); alert('送信に失敗しました。'); }) .finally(() => { submitBtn.disabled = false; submitBtn.textContent = '依頼を送信する'; }); });
